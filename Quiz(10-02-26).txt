5. import java.time.*;

class Test {
  public static void main(String[] args) {
    LocalDate start = LocalDate.of(2016, Month.FEBRUARY, 29);
    LocalDate end = LocalDate.of(2016, Month.MAY, 1);
    Period period = Period.ofMonths(1);

    sendAlerts(start, end, period);
  }

  private static void sendAlerts(LocalDate start,
    LocalDate end, Period period)
  {
    while (start.isBefore(end)) {
      System.out.println("Reminder " + start);
      start = start.plus(period);
    }
  }
}
Which dates are printed?

A. 2016-02-29, 2016-03-29, 2016-04-29
B. 2016-02-29, 2016-03-31, 2016-04-30
C. 2016-02-29, 2016-03-29, 2016-04-30
D. 2016-02-29, 2016-03-31, 2016-04-29

 Answer: A
